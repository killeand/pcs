CREATE TABLE dbo.player (
	player_id		INT				IDENTITY(1,1)	NOT NULL

CREATE PROCEDURE dbo.pr_player_new
	@first_name		VARCHAR(100)
	, @last_name	VARCHAR(100)
	, @email		VARCHAR(255)
	, @password		BINARY(128)
	, @salt			BINARY(128)
	, @address		VARCHAR(15)
AS
	INSERT INTO dbo.player (first_name, last_name, email, password, salt, address)
	VALUES (@first_name, @last_name, @email, @password, @salt, @address)
GO

CREATE PROCEDURE dbo.pr_player_group_new
	@player_id		INT
	, @acl_id		INT
AS
	INSERT INTO dbo.player_group (player_id, acl_id)
	VALUES (@player_id, @acl_id)
GO

CREATE PROCEDURE dbo.pr_player_delete
	@player_id		INT
AS
	DELETE FROM dbo.player
	WHERE player_id = @player_id
GO

CREATE PROCEDURE dbo.pr_player_change_details
	@player_id		INT
	, @first_name	VARCHAR(100)
	, @last_name	VARCHAR(100)
AS
	UPDATE dbo.player
	SET
		first_name = @first_name
		, last_name = @last_name
	WHERE player_id = @player_id
GO

CREATE PROCEDURE dbo.pr_player_change_email
	@player_id		INT
	, @email		VARCHAR(255)
AS
	UPDATE dbo.player
	SET	email = @email
	WHERE player_id = @player_id
GO

CREATE PROCEDURE dbo.pr_player_change_password
	@player_id		INT
	, @password		BINARY(128)
AS
	UPDATE dbo.player
	SET	password = @password
	WHERE player_id = @player_id
GO

CREATE PROCEDURE dbo.pr_player_change_address
	@player_id		INT

CREATE PROCEDURE dbo.pr_player_load_all

GO

CREATE PROCEDURE dbo.pr_player_load_by_id

GO

CREATE PROCEDURE dbo.pr_player_load_by_email
	@email			VARCHAR(255)

	SELECT * FROM dbo.player WHERE email = @email
GO

CREATE PROCEDURE dbo.pr_player_load_groups

GO

CREATE PROCEDURE dbo.pr_player_delete_groups
	@player_id		INT

	DELETE FROM dbo.player_group WHERE player_id = @player_id
GO

CREATE PROCEDURE dbo.pr_player_check_email
	@email			VARCHAR(255)
AS
	SELECT player_id FROM dbo.player WHERE email = @email
GO

CREATE PROCEDURE dbo.pr_player_is_my_character
	@player_id			INT
	, @character_id		BIGINT
AS
	SELECT * FROM dbo.player_character WHERE player_id = @player_id AND character_id = @character_id
GO

CREATE PROCEDURE dbo.pr_player_is_gm_of_character
	@player_id			INT
	, @character_id		BIGINT
AS
	DECLARE @campaign_id INT
	SELECT @campaign_id = campaign_id FROM dbo.player_character WHERE character_id = @character_id
	
	SELECT campaign_id FROM dbo.campaign WHERE player_id = @player_id AND campaign_id = @campaign_id
GO

CREATE PROCEDURE dbo.pr_acl_new
	@name				VARCHAR(255)
	, @description		VARCHAR(255)
	, @acl_bits			BINARY(8)
AS
	INSERT INTO dbo.acl_group (name, description, acl_bits)
	VALUES (@name, @description, @acl_bits)
GO

CREATE PROCEDURE dbo.pr_acl_update
	@acl_id				INT
	, @name				VARCHAR(255)
	, @description		VARCHAR(255)
	, @acl_bits			BINARY(8)
AS
	UPDATE dbo.acl_group
	SET
		name = @name
		, description = @description
		, acl_bits = @acl_bits
	WHERE acl_id = @acl_id
GO

CREATE PROCEDURE dbo.pr_acl_delete
	@acl_id		INT
AS
	DELETE FROM dbo.acl_group WHERE acl_id = @acl_id
	DELETE FROM dbo.player_group WHERE acl_id = @acl_id
GO

CREATE PROCEDURE dbo.pr_acl_load_by_id
	@acl_id		INT
AS
	SELECT * FROM dbo.acl_group WHERE acl_id = @acl_id
GO

CREATE PROCEDURE dbo.pr_acl_load_all
AS
	SELECT acl_id FROM dbo.acl_group
GO
